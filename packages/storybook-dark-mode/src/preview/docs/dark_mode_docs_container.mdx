# DarkModeDocsContainer

This component is used to render docs container with dark mode support.

## Usage

```tsx
// .storybook/preview.tsx
import { definePreview } from '@storybook/your-framework'
import { DarkModeDocsContainer } from '@storybook-community/storybook-dark-mode/docs'

export default definePreview({
  parameters: {
    docs: {
      container: DarkModeDocsContainer
    },
  }
})
```
